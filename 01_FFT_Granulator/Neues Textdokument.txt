function copyArray()
{
    newFreqs = [...freqs];
}

function init()
{	
	if(initialized)
		return;
		
	copyArray();
	initialized = true;
	console.log("Initialized");
}

{
    var tmp = [];

	for(i = 0; i < a; i++)
	{
		for(j = 0; j < newFreqs.length-1; j++)
		{
				tmp.push((newFreqs[j+1] - newFreqs[j]) * 0.5 + newFreqs[j]);	
		}
			for(k = 0; k < tmp.length; k++)
		{
				newFreqs.push(tmp[k]);	
		}
		newFreqs.sort(function(a, b){return a - b}); ;
		tmp = [];
	}
